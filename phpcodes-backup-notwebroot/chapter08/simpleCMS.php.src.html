  if (filter_has_var(INPUT_GET, "content")){
    $content = filter_input(INPUT_GET, "content");

    $content = "default.html";
  } // end if

  include ("top.html");

  print "<div class = \"menuPanel\"> \n";
  include ($menu);
  print "</div> \n";

  print "<div class = \"item\"> \n";
  include ($content);
  print "</div> \n";

function showPeople(){
  $formData = filter_input_array(INPUT_POST);
  $people = $formData["people"];

  print "<h2>Your People</h2> \n";
  print "<pre>";
  print_r($people);
  print "</pre>";
  
  print "<ul> \n";
  for ($i = 0; $i < count($people); $i++){
    print "  <li>$people[$i] </li> \n";
  } // end for loop
  print "</ul> \n";
} // end showPeople

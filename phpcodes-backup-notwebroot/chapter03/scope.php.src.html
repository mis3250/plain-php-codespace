<?php 

$a = "I have a value";
$b = "I have a value";

print <<<HERE
  <p>
  outside the function, <br />
  \$a is "$a", and<br />
  \$b is "$b"
  </p>
HERE;

myFunction();

function myFunction(){

  //make $a global, but not $b
  global $a;

  print <<<HERE
    <p>
    inside the function, <br />
    \$a is "$a", and<br />
    \$b is "$b"
		</p>
HERE;
} // end myFunction

?>
